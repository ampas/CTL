
if (TIFF_FOUND)

endif()

if ( OpenEXR_FOUND )
add_definitions( -DHAVE_OPENEXR=1 )

link_directories( ${OpenEXR_LIBRARY_DIRS} )
endif()

if ( AcesContainer_FOUND )
add_definitions( -DHAVE_ACESFILE=1 )

link_directories( ${AcesContainer_LIBRARY_DIRS} )

  transform.cc
  usage.cc
  aces_file.cc
  dpx_file.cc
  adx_file.cc
  exr_file.cc
  tiff_file.cc
  format.cc
  compression.cc

if (OpenEXR_FOUND)
